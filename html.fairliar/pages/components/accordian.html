<div class="accordian">
    <button class="acc-trigger"></button>

    <div class="acc-conts">
        <div class="inner">
            내용
        </div>
    </div>
</div>

MARKUP

이중 아코디언일 경우 서브아코디언에 .sub_accordian 추가


<!-- 메인 아코디언 -->
<div class="accordian type-02">
    <button class="acc-trigger"></button>

    <div class="acc-conts">
        <div class="inner">

            <!-- 서브 아코디언 -->
            <div class="accordian sub_accordian type-03">
                <button class="acc-trigger"></button>

                <div class="acc-conts">
                    <div class="inner">
                        내용
                    </div>
                </div>
            </div>
            <!-- //서브 아코디언 -->

        </div>
    </div>
</div>
<!-- //메인 아코디언 -->
